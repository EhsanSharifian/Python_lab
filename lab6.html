class Set():
    def __init__(self):
        self.list = []

    def add(self, x):
        if x not in self.list:
            self.list.append(x)

    def delete(self, x):
        if x in self.list:
            self.list.remove(x)

    def __or__(self, other):
        union = Set()
        for member in self.list:
            union.add(member)
        for member in other.list:
            union.add(member)
        return union

    def __and__(self, other):
        intersect = Set()
        for member in self.list:
            if member in other.list:
                intersect.add(member)
        return intersect

    def __eq__(self, other):
        return set(self.list) == set(other.list)
